    Structure:
    ```swift
    import Testing
    import SwiftData
    import Foundation
    @testable import Travellify

    @MainActor
    struct PackingTests {

        private func makeContainer() throws -> ModelContainer {
            let config = ModelConfiguration(isStoredInMemoryOnly: true)
            return try ModelContainer(
                for: Trip.self, Destination.self, Document.self,
                     PackingItem.self, PackingCategory.self, Activity.self,
                configurations: config
            )
        }

        private func makeTrip(in context: ModelContext) -> Trip {
            let trip = Trip()
            trip.name = "Rome 2026"
            trip.startDate = Date()
            trip.endDate = Date().addingTimeInterval(86_400 * 5)
            context.insert(trip)
            return trip
        }
    }
    ```

    Required tests (each uses `try makeContainer().mainContext`):

    1. **`packingCategoryDefaults`** — Create a `PackingCategory()` with no field assignments; `#expect` id != empty UUID wrapper, `name == ""`, `sortOrder == 0`, `trip == nil`, `(items ?? []).isEmpty`.

    2. **`packingItemDefaults`** — Create a `PackingItem()`; `#expect` `name == ""`, `isChecked == false`, `sortOrder == 0`, `category == nil`.

    3. **`insertItemUnderCategoryRoundTrip`** (PACK-01, PACK-02) — Make trip, insert category with `trip = trip, sortOrder = 0`, insert item with `category = category, name = "Toothbrush", sortOrder = 0`, save. Fetch category back, `#expect` `category.items?.count == 1`, `category.items?.first?.name == "Toothbrush"`, `category.trip?.persistentModelID == trip.persistentModelID`.

    4. **`deleteCategoryCascadesToItems`** (PACK-04) — Seed one category with three items, save. `modelContext.delete(category)`, save. Fetch all `PackingItem` rows — `#expect(... .isEmpty, "Category delete must cascade to items")`. Also fetch `PackingCategory` — `#expect(... .isEmpty)`.

    5. **`deleteTripCascadesToCategoriesAndItemsTwoLevel`** (PACK-01, PACK-04 / redundant-but-worth-having alongside the TripTests version from plan 01) — Seed trip with two categories (2 items each). Delete trip, save. `#expect` all four items and both categories are gone. If plan 01's `deleteTripCascadesToPackingModels` already covers this with the same depth, mark this test as a more thorough multi-category version (keep both).

    6. **`isCheckedDefaultsFalseAndPersistsToggle`** (PACK-06) — Seed trip + category + item (don't touch isChecked). Save. Refetch item via `FetchDescriptor<PackingItem>(predicate: #Predicate { $0.id == uuid })`; `#expect(item.isChecked == false)`. Toggle to true, save, refetch, `#expect(item.isChecked == true)`. Toggle back, save, refetch, `#expect(item.isChecked == false)`.

    7. **`sortOrderMonotonicOnInsert`** (PACK-01 / D21) — Insert 3 items into a category with `sortOrder = 0, 1, 2`. Save. Fetch back sorted by `sortOrder` ascending. `#expect` names come back in insertion order. (Validates the append-to-end sorting convention.)

    8. **`categoryWithoutItemsHasEmptyArrayNotNilAfterSave`** — Create category with no items, save, refetch. `#expect` that `(category.items ?? []).isEmpty` is true. (Sanity check for the `[PackingItem]? = []` default.)

    Do NOT add tests for UI behaviors (swipe, focus, drag-drop) — those are outside Swift Testing's scope on this project (UI tests live in XCTest per CLAUDE.md). This file covers data-layer invariants only.

    Helper pattern (shared across tests):
    ```swift
    private func seedCategory(name: String, itemStates: [(String, Bool)], in context: ModelContext, trip: Trip) -> PackingCategory {
        let cat = PackingCategory()
        cat.name = name
        cat.sortOrder = 0
        cat.trip = trip
        context.insert(cat)
        for (idx, (itemName, checked)) in itemStates.enumerated() {
            let item = PackingItem()
            item.name = itemName
            item.isChecked = checked
            item.sortOrder = idx
            item.category = cat
            context.insert(item)
        }
        return cat
    }
    ```

    2. Inserts a trip + seeded categories.
    3. Saves.
    4. Reads `trip.packingCategories ?? []` and applies the formula.
    5. Asserts.
