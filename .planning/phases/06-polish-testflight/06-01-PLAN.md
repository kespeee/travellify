    ```swift
    @MainActor
    static func nextDefaultName(in trip: Trip) -> String {
        let regex = /^doc-(\d+)$/
        let maxN = (trip.documents ?? [])
            .compactMap { doc -> Int? in
                guard let match = try? regex.wholeMatch(in: doc.displayName),
                      let n = Int(match.output.1) else { return nil }
                return n
            }
            .max() ?? 0
        return "doc-\(maxN + 1)"
    }
    ```
