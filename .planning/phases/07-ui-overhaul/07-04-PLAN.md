    ```swift
    import SwiftUI

    /// Packing list item row matching Figma node 157:3781/157:3867.
    /// Three render modes via Mode enum: existing item, add-item placeholder.
    struct PackingItemRow: View {
        enum Mode {
            case item(PackingItem)
            case addPlaceholder
        }

        let mode: Mode
        let onToggle: () -> Void
        let onCommitNewItem: (String) -> Void
        let onTapLabel: () -> Void   // request inline rename (parent handles state)

        @State private var draftText: String = ""
        @FocusState private var isEditingDraft: Bool

        var body: some View {
            HStack(spacing: 8) {
                checkbox
                label
            }
            .frame(height: 40)
            .contentShape(Rectangle())
        }

        @ViewBuilder
        private var checkbox: some View {
            switch mode {
            case .item(let item):
                Button(action: onToggle) {
                    if item.isChecked {
                        ZStack {
                            RoundedRectangle(cornerRadius: 8, style: .continuous)
                                .fill(Color.accentColor)
                            Image(systemName: "checkmark")
                                .font(.footnote.weight(.semibold))
                                .foregroundStyle(.white)
                        }
                    } else {
                        RoundedRectangle(cornerRadius: 8, style: .continuous)
                            .fill(Color(.secondarySystemBackground))
                            .overlay {
                                RoundedRectangle(cornerRadius: 8, style: .continuous)
                                    .strokeBorder(Color(.separator), lineWidth: 1)
                            }
                    }
                }
                .frame(width: 24, height: 24)
                .buttonStyle(.plain)
            case .addPlaceholder:
                RoundedRectangle(cornerRadius: 8, style: .continuous)
                    .fill(Color(.secondarySystemBackground))
                    .overlay {
                        RoundedRectangle(cornerRadius: 8, style: .continuous)
                            .strokeBorder(Color(.separator), style: StrokeStyle(lineWidth: 1, dash: [4, 2]))
                    }
                    .frame(width: 24, height: 24)
            }
        }

        @ViewBuilder
        private var label: some View {
            switch mode {
            case .item(let item):
                Text(item.name)
                    .font(.headline.weight(.semibold))
                    .foregroundStyle(item.isChecked ? Color(red: 0x72/255, green: 0x72/255, blue: 0x72/255) : .primary)
                    .strikethrough(item.isChecked)
                    .frame(maxWidth: .infinity, alignment: .leading)
                    .contentShape(Rectangle())
                    .onTapGesture { onTapLabel() }
                    .animation(.easeInOut(duration: 0.2), value: item.isChecked)
            case .addPlaceholder:
                TextField("Add item", text: $draftText)
                    .font(.headline.weight(.semibold))
                    .foregroundStyle(.primary)
                    .focused($isEditingDraft)
                    .submitLabel(.done)
                    .onSubmit {
                        let trimmed = draftText.trimmingCharacters(in: .whitespaces)
                        if !trimmed.isEmpty {
                            onCommitNewItem(trimmed)
                            draftText = ""
                            isEditingDraft = true   // keep focus for next item (chained adds)
                        } else {
                            isEditingDraft = false
                        }
                    }
                    .frame(maxWidth: .infinity, alignment: .leading)
            }
        }
    }
    ```

    **Notes:**
    - Tap on the label triggers `onTapLabel()` — parent owns the inline-rename state (existing PackingRow pattern preserved at the parent).
    - `addPlaceholder` shows the placeholder text via TextField's prompt, but uses real text input. Once user types + commits, `onCommitNewItem(text)` fires; placeholder text shows "Add item" as TextField prompt. Use `.font(.headline.weight(.semibold))` so muted placeholder matches Figma.
    - On commit, keep focus active for chained-add UX (consistent with Phase 3 inline add).
    - Animate strikethrough via `.animation(.easeInOut, value: item.isChecked)` so checking transitions smoothly.
